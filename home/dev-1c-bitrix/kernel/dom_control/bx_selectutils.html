BX.selectUtils.addNewOption

Добавляет в существующий список новое значение. Параметры: 

//Добавить option  в select
BX.selectUtils.addNewOption(BX('id_select'), 'NAME', 'По наименованию');
BX.selectUtils.addNewOption(BX('id_select'), 'PRICE', 'По цене');

 

BX.selectUtils.selectOption(DOMNode node, string value)

Выделяет указанное значение указанного списка.

//Выделить option с значением NAME
BX.selectUtils.selectOption(createSelect, 'PRICE');

 

BX.selectUtils.selectAllOptions(DOMNode node)

Выделяет все значения указанного списка.

 

BX.selectUtils.deleteOption(DOMNode node, string value)

Удаляет из указанного списка значение с указанным значением. 

//Удаляет option с значением DATE
BX.selectUtils.deleteOption(BX('id_select'), 'DATE');

 

BX.selectUtils.deleteSelectedOptions(DOMNode node)

Удаляет из указанного списка все выделенные значения.

 

BX.selectUtils.deleteAllOptions(DOMNode node)

Удаляет из указанного списка все значения.

//Удаляет все options
BX.selectUtils.deleteAllOptions(BX('id_select'));

 

BX.selectUtils.sortSelect(DOMNode node)

Сортирует указанный список.

 

BX.selectUtils.moveOptionsUp(DOMNode node)

Перемещает выделенные значения указанного списка на одну позицию вверх.

 

BX.selectUtils.moveOptionsDown(DOMNode node)

Перемещает выделенные значения указанного списка на одну позицию вниз.

 

Примеры

//Создать узел
var selecList = BX.create('select');
//Добавить option  в select

BX.selectUtils.selectOption(selecList, 'NAME');
//Выставить в дом собранный узел select
BX('sort-panel').append(selecList);

 

<script>
BX.ready(function() {
	BX.bind(BX('togglediv'), 'click', function() {
		//Создать узел
		var createSelect = BX.create('select', {
			props: {
				id: "id_select",
				size: 5
			}
		});
		//Добавить option  в select
		BX.selectUtils.addNewOption(createSelect, 'NAME', 'По наименованию');
		BX.selectUtils.addNewOption(createSelect, 'PRICE', 'По цене');
		//Выделить option с значением NAME
		BX.selectUtils.selectOption(createSelect, 'PRICE');
		//Выставить в дом собранный узел select
		BX('result').append(createSelect);
		BX('optn_result').style.display = "block";
		BX('togglediv').style.display = "none";
	});
	BX.bind(BX('addNewOptn'), 'click', function() {
		//Добавить option  в select
		BX.selectUtils.addNewOption(BX('id_select'), 'DATE', 'По дате');
		//Выделить option с значением DATE
		BX.selectUtils.selectOption(BX('id_select'), 'DATE');
		BX('deleteOptn').style.display = "block";
	});
	BX.bind(BX('deleteOptn'), 'click', function() {
		//Удаляет option с значением DATE
		BX.selectUtils.deleteOption(BX('id_select'), 'DATE');
	});
	BX.bind(BX('deleteAllOptn'), 'click', function() {
		//Удаляет все options
		BX.selectUtils.deleteAllOptions(BX('id_select'));
	});
});
</script>
