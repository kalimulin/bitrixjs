BX.setOpacity

BX.setOpacity(
 DOMNode node,
 integer percentage
);

Метод устанавливает уровень прозрачности элемента веб-страницы.

 

Параметры функции

 

Пример

<script>
BX.ready(function(){
 BX.bind(BX('buttonDivId'), 'click', function() {
 //Метод устанавливает уровень прозрачности элемента веб-страницы
  BX.setOpacity(this, 55); //BX.setOpacity(BX('buttonDivId'), 55);
 }); 
});
</script>

<div id="buttonDivId" class="buttonDiv">setOpacity 55%</div>
