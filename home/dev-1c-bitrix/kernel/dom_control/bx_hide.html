Скрытие элемента DOM:

BX.hide(element);

Пример

for (let element of document.querySelectorAll('.price_type')) {//обход каждого элемента с классом
  BX.hide(element);//Скрытие всего по умолчанию при выборе option
}

 

Отображение скрытого элемента DOM

BX.show(element);

Пример

for (let element of document.querySelectorAll('.price_type')) {
  BX.show(element);
}

 

Переключение режима отображения/скрытия элемента

Скрывает отображаемый или отображает скрытый элемент

BX.toggle(DOMNode node)

 

Определить, скрыт ли указанный элемент

BX.isNodeHidden(
 DOMNode node
);

Пример

<script>
 BX.ready(function(){
    BX.bind(BX('togglediv'), 'click', function() {
     BX.adjust(BX('result'), {
      style: {display: "block"},
      text: " t1  "+BX.isNodeHidden(BX('t1'))+" t2  "+BX.isNodeHidden(BX('t2'))
      });
    });    
 });
</script>

<div id="togglediv" class="arrowUp">Whether the node is hidden</div>
<input id="t1" type="text" maxlength="20" value="t1">
<input id="t2" type="hidden" maxlength="20" value="t2">
<div id="result" style="display:none" ></div>
