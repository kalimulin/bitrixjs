BX.remove

null 
BX.remove(
 DOMNode node
);

Функция удаляет узел DOM-структуры.

 

Примеры

<div id="togglediv" class="arrowGreen"> <span id="id_img">toggleClass</span></div>

<script>
BX.ready(function() {
	BX.bind(BX('togglediv'), 'click', function() {
		//Функция переключает наличие/отсутствие CSS-класса className у узла node
		BX.toggleClass(this, Array("arrowUp", "arrowDown", "arrowGreen"));
		BX.remove(BX('id_img'))
		textclass = this.getAttribute('class');
		this.appendChild(BX.create("img", {
			props: {
				id: "id_img",
				src: "/razrabs/JS/dom/img/" + textclass + ".gif",
				alt: 'togglediv'
			}
		}));
	});
});
</script>
