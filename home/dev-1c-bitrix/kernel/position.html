BX.GetWindowInnerSize

Получение текущего просматриваемого размера окна.

<script>
BX.ready(function() {
	BX.bind(BX('buttonfalse'), 'click', function() {
		//задает) ширину
		BX.width(this, 400); //BX.width(BX('buttonDivId'), 300);
		var objpos = BX.GetWindowInnerSize();
		this.innerHTML = "innerHeight " + objpos.innerHeight + "<br /> innerWidth " + objpos.innerWidth;
	});
});
</script>

<div id="buttonfalse" class="buttonDiv">set GetWindowInnerSize</div>

<style>

	cursor: pointer;
	margin: 12px auto;
	vertical-align: middle;
	text-align: center;
	width: 250px;
	background-color: #c5c5c5;
	border-radius: 5px;
	box-shadow: 0px 0px 10px #333;
	color: #000;
	font-size: 17px;
}
</style>

BX.pos

Получение абсолютной позиции элемента на экране.

Возвращает объект с ключами top, left, width, bottom, right, height.

Если указать bRelative=true, то будут считаться относительно ближайшего родителя с абсолютным или относительным позиционированием.

Примечание: Может некорректно работать для элементов с фиксированной позицией (position: fixed) или для их дочерних элементов.

Object
BX.pos(
 DOMNode node,
 bool bRelative
);
<script>
BX.ready(function() {
	BX.bind(BX('buttonDivId'), 'click', function() {
		//задает) ширину
		BX.width(this, 100); //BX.width(BX('buttonDivId'), 300);
		var objpos = BX.pos(this, bRelative = true);
		this.innerHTML = "top  " + objpos.top + "<br />left  " + objpos.left + "<br />width  " + objpos.width + "<br />bottom  " + objpos.bottom + "<br />right  " + objpos.right + "<br />height  " + objpos.height;
	});
	BX.bind(BX('buttonfalse'), 'click', function() {
		//задает) ширину
		BX.width(this, 100); //BX.width(BX('buttonDivId'), 300);
		var objpos = BX.pos(this, bRelative = false);
		this.innerHTML = "top  " + objpos.top + "<br />left  " + objpos.left + "<br />width  " + objpos.width + "<br />bottom  " + objpos.bottom + "<br />right  " + objpos.right + "<br />height  " + objpos.height;
	});
});
</script>
