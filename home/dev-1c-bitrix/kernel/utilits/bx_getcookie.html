BX.getCookie(
 name
);

Возвращает значение cookie с именем name.

 

Параметры функции

 

Примеры использования

document.cookie = 'test=123456';
BX.getCookie('test'); // вернет - 123456 
//отслеживание входа в личный кабинет (авторизация)
if (BX.getCookie('BITRIX_SM_USER_AUTH')) {
  if (typeof dataLayer == 'undefined') {
    dataLayer = [{
      'event' : 'GAEvent',
      'eventCategory' : 'form',
      'eventAction' : 'success',
      'eventLabel' : 'log_in'
    }];
  } else {
    dataLayer.push({
      'event' : 'GAEvent',
      'eventCategory' : 'form',
      'eventAction' : 'success',
      'eventLabel' : 'log_in'
    });
  }   
	
  // после выполнения удаляем куку
  BX.setCookie('BITRIX_SM_USER_AUTH', '', {path: '/', expires: -100});
}
