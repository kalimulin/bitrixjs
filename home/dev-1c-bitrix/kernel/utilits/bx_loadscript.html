Загрузить и выполнить скрипт или серию скриптов. Скрипты будут загружены средствами DOM, без XMLHttpRequest.

void 
BX.loadScript(
 string|array script[,
 function callback[
 DOMDocument doc]]
);

Функция загрузит и выполнит скрипт или серию скриптов. Скрипты будут загружены средствами DOM, без XMLHttpRequest. Если передан массив скриптов и обработчик, то обработчик будет вызван после загрузки всех скриптов

 

Параметры функции

 

Пример

BX.loadScript('/bitrix/components/bitrix/main.colorpicker/templates/.default/script.js', function() {
  ...
});
BX.loadScript([
  "/bitrix/js/fileman/medialib/common.js?v=<?=@filemtime($_SERVER['DOCUMENT_ROOT'].'/bitrix/js/fileman/medialib/common.js')?>",
  "/bitrix/js/fileman/medialib/core_admin.js?v=<?=@filemtime($_SERVER['DOCUMENT_ROOT'].'/bitrix/js/fileman/medialib/core_admin.js')?>"
],
function() {
  ...
});
